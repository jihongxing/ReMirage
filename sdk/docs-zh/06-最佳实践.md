```go

    New: func() interface{} {
        client, _ := mirage.NewClient(endpoint, mirage.WithToken(token))
        return client
    },
}

client := pool.Get().(*mirage.Client)
defer pool.Put(client)
```

## 心跳策略

### 自适应间隔

```python
class HeartbeatManager:
    def __init__(self, client):
        self.client = client
        self.interval = 30  # 默认 30 秒
    
    def run(self):
        while True:
            response = self.client.gateway.sync_heartbeat(request)
            # 使用服务端返回的间隔
            self.interval = response.next_heartbeat_interval
            time.sleep(self.interval)
```

### 错过心跳处理

```python
# 连续 3 次心跳失败，触发重连
consecutive_failures = 0
MAX_FAILURES = 3

while True:
    try:
        response = client.gateway.sync_heartbeat(request)
        consecutive_failures = 0
    except Exception:
        consecutive_failures += 1
        if consecutive_failures >= MAX_FAILURES:
            # 重新初始化连接
            client = MirageClient(endpoint, token)
            consecutive_failures = 0
    time.sleep(30)
```

## 流量上报

### 批量上报

```python
# 累积 10 秒数据后上报
class TrafficReporter:
    def __init__(self, client):
        self.client = client
        self.base_bytes = 0
        self.defense_bytes = 0
    
    def add(self, base, defense):
        self.base_bytes += base
        self.defense_bytes += defense
    
    def flush(self):
        if self.base_bytes > 0 or self.defense_bytes > 0:
            self.client.gateway.report_traffic(TrafficReport(
                gateway_id=GATEWAY_ID,
                base_traffic_bytes=self.base_bytes,
                defense_traffic_bytes=self.defense_bytes
            ))
            self.base_bytes = 0
            self.defense_bytes = 0
```

## 配额管理

### 预警阈值

```python
def check_quota(response):
    remaining_percent = response.remaining_bytes / response.total_bytes * 100
    
    if remaining_percent < 5:
        # 紧急：暂停非必要流量
        pause_non_essential()
    elif remaining_percent < 10:
        # 警告：通知管理员
        notify_admin()
    elif remaining_percent < 20:
        # 提醒：考虑购买
```

```python
def auto_renew(client, account_id, threshold_percent=10):
    balance = client.billing.get_balance(account_id)
    remaining_percent = balance.remaining_bytes / balance.total_bytes * 100
```

```python
def handle_threat(threat_data):
    severity = threat_data['severity']
    
    if severity >= 9:
        # 紧急：立即切换蜂窝
        client.cell.switch_cell(SwitchCellRequest(
            reason=SwitchReason.THREAT_DETECTED
        ))
    elif severity >= 7:
        # 高危：提升防御等级
        # 由服务端自动处理
        pass
    elif severity >= 5:
        # 中危：记录并监控
        log_threat(threat_data)
    else:
        # 低危：仅记录
        log_info(threat_data)
```

## 蜂窝选择

### 就近原则

```python
def select_best_cell(cells, user_country):
    # 1. 优先同国家
    same_country = [c for c in cells if c.country == user_country]
    if same_country:
        cells = same_country
    
    # 2. 选择负载最低
    cells.sort(key=lambda c: c.load_percent)
    
    # 3. 排除高负载
    cells = [c for c in cells if c.load_percent < 80]
    
    return cells[0] if cells else None
```
